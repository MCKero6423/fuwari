```python
class TodoManager:
    def __init__(self):
        self.tasks = []
    
    def add_task(self, task, priority="medium"):
        """添加新任务"""
        self.tasks.append({
            "task": task,
            "priority": priority,
            "completed": False,
            "created_date": datetime.now()
        })
        print(f"任务 '{task}' 已添加")
    
    def complete_task(self, task_index):
        """标记任务为完成"""
        if 0 <= task_index < len(self.tasks):
            self.tasks[task_index]["completed"] = True
            print("任务已完成！")
        else:
            print("任务索引无效")
```
